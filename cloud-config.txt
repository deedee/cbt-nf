#cloud-config

package_update: true     # Update the apt database
package_upgrade: true   # Upgrade the instance's packages

power_state:
  delay: "now"
  mode: reboot
  message: Reboot after installing Docker
  condition: True

# Cloud-Config begins
write_files:
  - path: /etc/systemd/system/docker-compose-autostart.service
    permissions: '0644'
    content: |
      [Unit]
      Description=Docker Compose Auto-Start for Geonode
      After=docker.service
      Requires=docker.service

      [Service]
      Restart=always
      WorkingDirectory=/opt/cbt-nf
      ExecStart=/usr/local/bin/docker-compose up -d
      ExecStop=/usr/local/bin/docker-compose down
      TimeoutStartSec=0

      [Install]
      WantedBy=multi-user.target

  - apt-get remove -y docker docker-engine docker.io containerd runc

  - apt-get install -y \
        ca-certificates \
        curl \
        gnupg \
        lsb-release
  - mkdir -p /etc/apt/keyrings
  - curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /etc/apt/keyrings/docker.gpg
  - echo \
      "deb [arch=$(dpkg --print-architecture) signed-by=/etc/apt/keyrings/docker.gpg] https://download.docker.com/linux/ubuntu $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null

  # 4. Clone the GitHub repo

  # 5. Enable and start the docker-compose service on boot

  - systemctl start docker-compose-autostart.service
