#cloud-config
# Swap size will be twice the amount of actual memory (RAM)
# Assuming the system has a default volume where the swap file can be created

# Variables
SWAP_FILE=/swap.bin
REPO_URL=https://github.com/deedee/cbt-nf
REPO_DIR=/opt/cbt-nf

# Cloud-Config begins
write_files:
  - path: /etc/systemd/system/docker-compose-autostart.service
    permissions: '0644'
    content: |
      [Unit]
      Description=Docker Compose Auto-Start for Geonode
      After=docker.service
      Requires=docker.service

      [Service]
      Restart=always
      WorkingDirectory=/opt/cbt-nf
      ExecStart=/usr/local/bin/docker-compose up -d
      ExecStop=/usr/local/bin/docker-compose down
      TimeoutStartSec=0

      [Install]
      WantedBy=multi-user.target

runcmd:
  # 1. Determine the amount of memory and create swap file of size 2x memory
  - |
    MEMORY=$(grep MemTotal /proc/meminfo | awk '{print $2}')
    SWAP_SIZE_MB=$((MEMORY * 2 / 1024))
    fallocate -l "${SWAP_SIZE_MB}M" $SWAP_FILE
    chmod 600 $SWAP_FILE
    mkswap $SWAP_FILE
    echo "$SWAP_FILE none swap sw 0 0" >> /etc/fstab
    swapon -a
  
  # 2. Install updates
  - apt-get update && apt-get -y upgrade

  # 3. Install latest Docker CE
  - apt-get remove -y docker docker-engine docker.io containerd runc
  - apt-get update
  - apt-get install -y \
        ca-certificates \
        curl \
        gnupg \
        lsb-release
  - mkdir -p /etc/apt/keyrings
  - curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /etc/apt/keyrings/docker.gpg
  - echo \
      "deb [arch=$(dpkg --print-architecture) signed-by=/etc/apt/keyrings/docker.gpg] https://download.docker.com/linux/ubuntu \
      $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null
  - apt-get update
  - apt-get install -y docker-ce docker-ce-cli containerd.io docker-buildx-plugin docker-compose-plugin

  # 4. Clone the GitHub repo
  - git clone $REPO_URL $REPO_DIR

  # 5. Enable and start the docker-compose service on boot
  - systemctl enable docker-compose-autostart.service
  - systemctl start docker-compose-autostart.service

  # 6. Reboot if required by upgrades
  - if [ -f /var/run/reboot-required ]; then reboot; fi
